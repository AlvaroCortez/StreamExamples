java.util.stream.Stream<T>
public abstract <R> Stream<R> flatMap(java.util.function.Function<? super T, ? extends Stream<? extends R>> mapper)

Examples. 


     List<Order> orders = asList(
            new Order("food", asList("apple", "orange")),
            new Order("clothes", asList("shirt", "dress")));

     List<String> itemNames = orders.stream()
            .flatMap(order -> order.getItems().stream())
            .collect(Collectors.toList());

     System.out.println(itemNames);  //output ["apple", "orange", "shirt", "dress"]
