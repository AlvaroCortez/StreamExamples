public abstract java.util.stream.DoubleStream flatMapToDouble(java.util.function.Function<? super T, ? extends DoubleStream> mapper)

Examples. 


     List<List<Double>> listOfDoubleLists = asList(asList(123.4, 56.78), asList(9.0, 12.34));

     double[] allDoubles = listOfDoubleLists.stream()
            .flatMapToDouble(doubleList -> doubleList.stream().mapToDouble(Double::doubleValue))
            .toArray();

     System.out.println(allDoubles);  // [123.4, 56.78, 9.0, 12.34]
