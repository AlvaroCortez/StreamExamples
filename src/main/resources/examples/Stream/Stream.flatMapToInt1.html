java.util.stream.Stream<T>
public abstract java.util.stream.IntStream flatMapToInt(java.util.function.Function<? super T, ? extends IntStream> mapper)

Examples. 


     List<Country> countries = asList(
                new Country("Russia", asList("Moscow", "Saint-Petersburg")),
                new Country("Great Britain", asList("London")));

     int[] citiesLength = cities.stream()
            .mapToInt(city -> city.length())
            .toArray();
     int[] allCitiesLength = countries.stream()
            .flatMapToInt(country -> country.cities.stream().mapToInt(String::length))
            .toArray();

     System.out.println(allCitiesLength);  // [6, 16, 6]
