java.util.stream.Stream<T>
public abstract boolean anyMatch(java.util.function.Predicate<? super T> predicate)

Examples. 


     List<Integer> numbers = asList(1, 2, 3, 4, 5);

     boolean anyNumberGraterThanSix = numbers.stream().anyMatch(number -> number > 6);

     System.out.println(anyNumberGraterThanSix);  // false

     boolean anyNumberEven = numbers.stream().anyMatch(number -> number & 2 == 0);

     System.out.println(anyNumberEven);  // true
