java.util.stream.Stream<T>
public abstract boolean noneMatch(java.util.function.Predicate<? super T> predicate)

Examples. 


     List<Integer> numbers = asList(1, 2, 3, 4, 5);

     boolean noneNumberGraterThanSix = numbers.stream().noneMatch(number -> number > 6);

     System.out.println(noneNumberGraterThanSix);  // true

     boolean noneNumberEven = numbers.stream().noneMatch(number -> number & 2 == 0);

     System.out.println(noneNumberEven);  // false
