java.util.stream.Stream<T>
public abstract boolean allMatch(java.util.function.Predicate<? super T> predicate)

Examples. 


     List<Integer> numbers = asList(1, 2, 3, 4, 5);

     boolean allNumbersGraterThanZero = numbers.stream().allMatch(number -> number > 0);

     System.out.println(allNumbersGraterThanZero);  // true

     boolean allNumbersEven = numbers.stream().allMatch(number -> number & 2 == 0);

     System.out.println(allNumbersEven);  // false
