java.util.stream.Stream<T>
public abstract <R> Stream<R> map(java.util.function.Function<? super T, ? extends R> mapper)

Examples. 


     List<Order> orders = asList(
            new Order("food", asList("apple", "orange")),
            new Order("clothes", asList("shirt", "dress")));

     List<String> orderNames = orders.stream()
            .flatMap(order -> order.getName())
            .collect(Collectors.toList());

     System.out.println(orderNames);  // ["food", "clothes"]
